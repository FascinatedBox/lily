                      LIBRARY_OUTPUT_DIRECTORY "${PROJECT_BINARY_DIR}/test/")

target_link_libraries(pre-commit-tests m)

if(LILY_NEED_DL)
    target_link_libraries(pre-commit-tests dl)
endif()
